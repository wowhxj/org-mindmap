* Test Case 1: Empty Root, Right-sided :layout top
#+begin_sexp
("" nil
 ("Node A" right)
 ("Node B" right ("Node C" right) ("Node D" right))
 ("Node E" right))
#+end_sexp
#+begin_expected
«» ┬─ Node A
   ├─ Node B ┬─ Node C
   │         ╰─ Node D
   ╰─ Node E
#+end_expected

* Test Case 2: Explicit Root, Left-sided :layout top
#+begin_sexp
("Root" nil
 ("Node A" left)
 ("Node B" left ("Node C" left) ("Node D" left))
 ("Node E" left))
#+end_sexp
#+begin_expected
          Node A ─┬ « Root »
Node C ─┬ Node B ─┤
Node D ─╯         │
          Node E ─╯
#+end_expected
